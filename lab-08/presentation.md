# –í—ã–ø–æ–ª–Ω–µ–Ω–∏–µ —Ä–∞–±–æ—Ç—ã:

```
import random
import string

inpt_1 = input("–í–≤–µ–¥–∏—Ç–µ –ø–µ—Ä–≤–æ–µ —Å–æ–æ–±—â–µ–Ω–∏–µ: ") #–ü—Ä–∏–≤–µ—Ç!
inpt_2 = input("–í–≤–µ–¥–∏—Ç–µ –≤—Ç–æ—Ä–æ–µ —Å–æ–æ–±—â–µ–Ω–∏–µ: ") #–®—Ç–∏—Ä–ª–∏—Ü
```

# –í—ã–ø–æ–ª–Ω–µ–Ω–∏–µ —Ä–∞–±–æ—Ç—ã:
–ù–∞–ø–∏—Å–∞–ª–∏ —Ñ—É–Ω–∫—Ü–∏—é –≥–µ–Ω–µ—Ä–∞—Ü–∏–∏ —Å–ª—É—á–∞–π–Ω–æ–≥–æ –∫–ª—é—á–∞, –∑–∞–¥–∞–Ω–Ω–æ–≥–æ —Ä–∞–∑–º–µ—Ä–∞. –ù–∞–ø–∏—Å–∞–ª–∏ —Ñ—É–Ω–∫—Ü–∏—é, –ø—Ä–µ–æ–±—Ä–∞–∑—É—é—â—É—é –∏—Å—Ö–æ–¥–Ω—ã–π —Å–∏–º–≤–æ–ª—å–Ω—ã–π —Ñ–æ—Ä–º–∞—Ç –≤ —à–µ—Å—Ç–Ω–∞–¥—Ü–∞—Ç–µ—Ä–∏—á–Ω—ã–π.
```
def key_gener(size = 6, chars = string.ascii_letters + string.digits):
    return ''.join(random.choice(chars) for _ in range(size))
def chan(s):
    return ":".join ("{:02x}".format(ord(c)) for c in s)
```

---
# –í—ã–ø–æ–ª–Ω–µ–Ω–∏–µ —Ä–∞–±–æ—Ç—ã:
–°–≥–µ–Ω–µ—Ä–∏—Ä–æ–≤–∞–ª–∏ —Å–ª—É—á–∞–π–Ω—ã–π –∫–ª—é—á, —Ä–∞—Å–ø–µ—á–∞—Ç–∞–ª–∏ –µ–≥–æ
```
key = key_gener(len(inpt_1))
ch_key = chan(key)
print(f'–ö–ª—é—á: {ch_key}') #–ö–ª—é—á: 70:4f:58:5a:61:5a:6e
```

---
# –í—ã–ø–æ–ª–Ω–µ–Ω–∏–µ —Ä–∞–±–æ—Ç—ã:
–°–æ–∑–¥–∞–ª–∏ —Ñ—É–Ω–∫—Ü–∏—é –≥–∞–º–º–∏—Ä–æ–≤–∞–Ω–∏—è.
```
def gam(inpt, key):
    inpt_ascii = [ord(i) for i in inpt]
    key_ascii = [ord(i) for i in key]
    enc_str = ''.join(chr(s ^ k) for s, k, in zip(inpt_ascii, key_ascii))
    return enc_str
```

```
encrypted_inpt_1 = gam(inpt_1, key)

```

